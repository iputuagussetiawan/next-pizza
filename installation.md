1. First Setup 
 npx create-next-app@latest next-pizza --typescript --tailwind --eslint

2. Add Shadcn UI
https://ui.shadcn.com/docs/installation/next
npx shadcn@latest init
npx shadcn@latest add button

import "./globals.css";

const nunito = Nunito({

});
